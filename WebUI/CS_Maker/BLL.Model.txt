using System;
using System.Collections;
using System.Collections.Generic;
using System.Text;
using System.Data;
using System.Data.OleDb;
using WC.Model;
using WC.DAL;
using WC.IDAL;
using WC.Factory;

namespace WC.BLL
{
    public class Admin
    {
        /// <summary>
        /// 在业务逻辑层中设置一个静态只读Admin操作类的对象
        /// </summary>
        private static readonly IAdmin dal = WC.Factory.DALFactory.CreateAdminDAL();

        /// <summary>
        /// 封装业务层中Admin操作类的构造方法
        /// </summary>
        private Admin() { }

        /// <summary>
        /// 业务逻辑层中Admin操作类对象的静态生成方法
        /// </summary>
        /// <returns></returns>
        public static Admin Init()
        {
            return new Admin();
        }

        #region NHibernate 方法

        public void Add(AdminInfo com)
        {
            dal.Add(com);
        }

        public void Update(AdminInfo com)
        {
            dal.Update(com);
        }

        public void Delete(int id)
        {
            dal.Delete(id);
        }

        public AdminInfo GetById(int id)
        {
            return dal.GetById(id);
        }

        public IList GetAll(string where, string orderBy)
        {
            return dal.GetAll(where, orderBy);
        }

        #endregion
    }
}
